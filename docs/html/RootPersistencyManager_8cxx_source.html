156 void RootPersistencyManager::writeHitsCollections(const G4Event* anEvent, event::Event* outputEvent) {
157 
158  // Clear the hits from last event.
159  for (auto entry : outputHitsCollections_) {
160  entry.second->Clear("C");
161  }
162 
163  // Get the HC of this event.
164  G4HCofThisEvent* hce = anEvent->GetHCofThisEvent();
165  int nColl = hce->GetNumberOfCollections();
166 
167  // Loop over all hits collections.
168  for (int iColl = 0; iColl < nColl; iColl++) {
169 
170  // Get a hits collection and its name.
171  G4VHitsCollection* hc = hce->GetHC(iColl);
172  std::string collName = hc->GetName();
173 
174  // Get the target output collection.
175  TClonesArray* outputHitsColl = outputHitsCollections_[collName];
176 
177  // If the collection is not found in the output ROOT event, then a fatal error occurs!
178  if (!outputHitsColl) {
179  std::cerr << "ERROR: The output collection " << collName << " was not found!" << std::endl;
180  G4Exception("RootPersistencyManager::writeHitsCollections",
181  "",
182  FatalException,
183  "The output collection was not found.");
184  }
185 
186  if (dynamic_cast<G4TrackerHitsCollection*>(hc) != nullptr) {
187 
188  // Write G4TrackerHit collection to output SimTrackerHit collection.
189  G4TrackerHitsCollection* trackerHitsColl = dynamic_cast<G4TrackerHitsCollection*>(hc);
190  writeTrackerHitsCollection(trackerHitsColl, outputHitsColl);
191 
192  } else if (dynamic_cast<G4CalorimeterHitsCollection*>(hc) != nullptr) {
193 
194  G4CalorimeterHitsCollection* calHitsColl = dynamic_cast<G4CalorimeterHitsCollection*>(hc);
195  if (collName == EventConstants::ECAL_SIM_HITS) {
196  // Write ECal G4CalorimeterHit collection to output SimCalorimeterHit collection using helper class.
197  ecalHitIO_->writeHitsCollection(calHitsColl, outputHitsColl);
198  } else {
199  // Write generic G4CalorimeterHit collection to output SimCalorimeterHit collection.
200  writeCalorimeterHitsCollection(calHitsColl, outputHitsColl);
201  }
202  }
203 
204  // Add hits collection to output event.
205  outputEvent->add(collName, outputHitsColl);
206 
207  if (m_verbose > 1) {
208  std::cout << "[ RootPersistencyManager ] : Wrote " << outputHitsColl->GetEntriesFast() << " hits into " << collName << std::endl;
209  }
210  }
211 }

281 void RootPersistencyManager::setupHitsCollectionMap() {
282

285  for (auto entry : outputHitsCollections_) {
286  delete entry.second;

288  }

290 
291  // Create an output TClonesArray in the map for each registered HC.
292  G4SDManager* sdMgr = G4SDManager::GetSDMpointer();
293  G4HCtable* hcTable = sdMgr->GetHCtable();
294  int entries = hcTable->entries();
295  for (int i = 0; i < entries; i++) {
296  std::string sdName = hcTable->GetSDname(i);
297  std::string hcName = hcTable->GetHCname(i);
298  G4VSensitiveDetector* sd = sdMgr->FindSensitiveDetector(sdName);
299  if (dynamic_cast<CalorimeterSD*>(sd)) {
300  outputHitsCollections_[hcName] = new TClonesArray("event::SimCalorimeterHit", 500);
301  if (m_verbose > 1) {
302  std::cout << "[ RootPersistencyManager ] - Created SimCalorimeterHit HC " << hcName << std::endl;
303  }
304  } else if (dynamic_cast<TrackerSD*>(sd)) {
305  outputHitsCollections_[hcName] = new TClonesArray("event::SimTrackerHit", 50);
306  if (m_verbose > 1) {
307  std::cout << "[ RootPersistencyManager ] - Created SimTrackerHit HC " << hcName << std::endl;
308  }
309  }
310  }
